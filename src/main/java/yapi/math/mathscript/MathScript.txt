rem.mathscript
  input(F a, F b)
  i = 0
  while (a > b) {
    a = a - b
    i = i + 1
  }
  return a
  return b
  return i

sum.mathscript
  input(N s, N e, E ex)
  if (s > e) {
  	return 0
  	break()
  }
  l = e - s
  t = 0
  loop(l) {
    n = l + s
    t = t + eval(ex)
  }
  return t
  return s
  return e
  return ex
